48      /**
49       * Checks the response code of the OneBusAway API and return the JSON Object named "data".
50       * 
51       * @param reponse
52       *            the reponse of the OneBusAway API
53       * @return the JSON Object named "data"
54       * @throws JSONException
55       *             the response does not contains valid data
56       * @throws OneBusAwayException
57       *             the API returned an error code
58       */
59      public static JSONObject getServiceResponse(final String reponse) throws JSONException,
60              OneBusAwayException {
61  
62          final JSONObject jsonResult = new JSONObject(reponse);
63  
64          final int code = jsonResult.getInt("code");
65  
66          if (code != RESPONSE_CODE_SUCCESS) {
67              final String message = String.format("OneBusAway API error : code=%S, %s ", code,
68                      jsonResult.getString("text"));
69              throw new OneBusAwayException(message, null);
70          }
71          return jsonResult.getJSONObject("data");
72      }

93       *            second timestamp
94       * @return the date
95       */
96      public static Date dateFromTimestamps(final Long timestamp1, final Long timestamp2) {
97  
98          final Calendar calendar = Calendar.getInstance();
99          calendar.setTimeInMillis(timestamp1 + timestamp2 * 1000);
100         return calendar.getTime();

104     /**
105      * Gets stops from the "references" part of OneBusAway response.
106      * 
107      * @param jsonArray
108      *            jsonArray containing the references part of the responses.
109      * @param routes
110      *            list of routes @link {@link #getReferencedRoutes(JSONArray)}
111      * @return the list of stops
112      */
113     public static HashMap<String, Stop> getReferencedStops(final JSONArray jsonArray,
114             final HashMap<String, Route> routes) {
115 
116         HashMap<String, Stop> stops = null;
117 
118         if (jsonArray != null) {
119             stops = new HashMap<String, Stop>();
120             for (int i = 0; !jsonArray.isNull(i); i++) {
121                 final JSONObject jsonStop = jsonArray.optJSONObject(i);
122                 stops.put(jsonStop.optString("id"), convertJsonObjectToStop(jsonStop, routes));
123             }
124         }
125         return stops;
126     }

128     /**
129      * Gets routes from the "references" part of OneBusAway response.
130      * 
131      * @param jsonArray
132      *            jsonArray containing the references part of the responses.
133      * @return the list of routes
134      */
135     public static HashMap<String, Route> getReferencedRoutes(final JSONArray jsonArray) {
136 
137         HashMap<String, Route> routes = null;
138 
139         if (jsonArray != null) {
140             routes = new HashMap<String, Route>();
141             for (int i = 0; !jsonArray.isNull(i); i++) {
142                 final JSONObject jsonRoute = jsonArray.optJSONObject(i);
143                 routes.put(jsonRoute.optString("id"), convertJsonObjectToRoute(jsonRoute));
144             }
145         }
146         return routes;
147     }

149     /**
150      * Converts a JSONObject to a Stop object.
151      * 
152      * @param jsonStop
153      *            JSONObject describing a Stop
154      * @param routes
155      *            list of routes @link {@link #getReferencedRoutes(JSONArray)}
156      * @return the Stop object
157      * @throws JSONException
158      */
159     public static Stop convertJsonObjectToStop(final JSONObject jsonStop,
160             final HashMap<String, Route> routes) {
161 
162         final Stop stop = new Stop();
163         stop.setId(jsonStop.optString("id"));
164         stop.setLon(jsonStop.optDouble("lon"));
165         stop.setLat(jsonStop.optDouble("lon"));
166         stop.setDirection(jsonStop.optString("direction"));
167         stop.setName(jsonStop.optString("name"));
168         stop.setCode(jsonStop.optInt("code"));
169         final JSONArray jsonRoutes = jsonStop.optJSONArray("routeIds");
170         for (int i = 0; !jsonRoutes.isNull(i); i++) {
171             stop.getRoutes().add(routes.get(jsonRoutes.optString(i)));
172         }
173 
174         return stop;
175     }

177     /**
178      * Converts a JSONObject to a route object.
179      * 
180      * @param jsonRoute
181      *            JSONObject describing a Route
182      * @return the Route object
183      */
184     public static Route convertJsonObjectToRoute(final JSONObject jsonRoute) {
185 
186         final Route route = new Route();
187         route.setId(jsonRoute.optString("id"));
188         route.setLongName(jsonRoute.optString("longName"));
189         route.setShortName(jsonRoute.optString("shortName"));
190         route.setTextColor(jsonRoute.optString("textColor"));
191         route.setColor(jsonRoute.optString("color"));
192         route.setAgencyId(jsonRoute.optString("agencyId"));
193         route.setDescription(jsonRoute.optString("description"));
194         route.setType(jsonRoute.optInt("type"));
195 
196         return route;
197     }
