86          final String url = PROPS.getProperty("onebusaway.api.url");

155     /**
156      * Test method for {@link OneBusAwayService#getScheduleForStop(String)}.
157      * 
158      * @throws IOException
159      *             an error occurred
160      * @throws ParseException
161      *             The date is not in a valid format
162      */
163     public final void testGetScheduleForStop() throws IOException, ParseException {
164 
165         LOGGER.info("testGetScheduleForStop.start");
166 
167         StopSchedule schedule = null;
168         final Calendar calendar = Calendar.getInstance();
169         final SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
170         calendar.setTime(sdf.parse("2011-07-27"));
171         schedule = obaClient.getScheduleForStop("1_repbottb", calendar.getTime());
172 
173         assertNotNull("no schedule returned by the api", schedule);
174 
175         assertEquals("109 stop times should be returned by the api", 109, schedule.getStopTimes()
176                 .size());
177         assertEquals("the stop id should be 1_repbottb", "1_repbottb", schedule.getStop().getId());
178         assertEquals("the stop name should be République Pré Botté", "République Pré Botté",
179                 schedule.getStop().getName());
180         assertEquals("3 lines should be return for this stop", 3, schedule.getStop().getRoutes()
181                 .size());
182 
183         int cpt = 0;
184         for (final ScheduleStopTime stopTime : schedule.getStopTimes()) {
185             if (stopTime.getRoute().getShortName().equals("50")) {
186                 cpt++;
187             }
188         }
189         assertEquals("54 stop times should be returned by the api for line 50", 54, cpt);
190 
191         LOGGER.info("testGetScheduleForStop.end");
192     }
